function groupIntoRows(items,tol){

    if(!items.length) return [];

    items.sort((a,b)=>a.y-b.y||a.x-b.x);

    const rows=[]; let cr=[items[0]];

    for(let i=1;i<items.length;i++){

        if(Math.abs(items[i].y-cr[0].y)<=tol) cr.push(items[i]);

        else{cr.sort((a,b)=>a.x-b.x);rows.push(cr);cr=[items[i]];}

    }

    cr.sort((a,b)=>a.x-b.x);rows.push(cr);return rows;

}

function parseFmt(s){if(!s)return 0;const n=parseFloat(s.replace(/,/g,''));return isNaN(n)?0:n;}

function detectIR2Type(allRows){

    const allText=allRows.map(r=>r.map(i=>i.text).join(' ')).join('\n');

    if(/\/IR2\//i.test(allText))return'ir2';

    if(/\/ACT\//i.test(allText))return'act';

    if(/\/H1\//i.test(allText))return'h1';

    if(/\/H2\//i.test(allText))return'h2';

    if(/\/D1\//i.test(allText))return'd1';

    if(/\/D2\//i.test(allText))return'd2';

    if(/\/A1\//i.test(allText))return'a1';

    if(/\/B1\//i.test(allText))return'b1';

    if(/\/D\//i.test(allText))return'd';

    if(/\/E\//i.test(allText))return'e';

    if(/\/G\//i.test(allText))return'g';

    if(/\/J\//i.test(allText))return'j';

    return'unknown';

}


function extractMeta(allRows){

    const r={};

    for(const row of allRows){

        const rt=row.map(i=>i.text).join(' ');

        const rec=rt.match(/(\d+-\d+-\d+-\d+\/\w+\/\d+)/);

        if(rec)r.numero_recepcion=rec[1];

        for(const it of row){

            if(/^\d{9}$/.test(it.text.trim())&&it.x<200&&!r.rnc){

                r.rnc=it.text.trim();

                // Buscar razón social en la misma fila (el texto largo después del RNC)

                const afterRnc=row.filter(x=>x.x>it.x+50).sort((a,b)=>a.x-b.x);

                for(const af of afterRnc){

                    const t=af.text.trim();

                    if(t.length>5&&!/^(Nombre|Apellido|Razón|RAZON|RNC|Cédula)/i.test(t)){r.razon_social=t;break;}

                }

            }

            if(/^\d{6}$/.test(it.text.trim())&&!r.periodo)r.periodo=it.text.trim();

        }

    }

    return r;

}

async function extractFromPDF(file){

    const ab=await file.arrayBuffer();

    const pdf=await pdfjsLib.getDocument({data:ab}).promise;

    let debugLog='', allRows=[];


    for(let p=1;p<=pdf.numPages;p++){

        const page=await pdf.getPage(p);

        const tc=await page.getTextContent();

        const vp=page.getViewport({scale:1.0});

        const items=tc.items.map(i=>({text:i.str,x:i.transform[4],y:vp.height-i.transform[5]})).filter(i=>i.text.trim()!=='');

        const rows=groupIntoRows(items,5);

        allRows.push(...rows);

        debugLog+=`\n=== PÁGINA ${p} ===\n`;

        for(const row of rows)debugLog+=`Y=${Math.round(row[0].y)}: ${row.map(i=>`[x=${Math.round(i.x)}]"${i.text}"`).join(' ')}\n`;

    }


    const allText=allRows.map(r=>r.map(i=>i.text).join(' ')).join('\n');

    let formType='unknown';

    if(/\/R17\//i.test(allText))formType='ir17';

    else if(/\/ITA\//i.test(allText))formType='anexoa';

    else if(/\/IT1\//i.test(allText))formType='it1';

    else if(/\/IR3\//i.test(allText))formType='ir3';

    else{

        const ir2Type=detectIR2Type(allRows);

        if(ir2Type!=='unknown')formType=ir2Type;

        else if(/RETRIBUCIONES\s+COMPLEMENTARIAS/i.test(allText))formType='ir17';

        else if(/OPERACIONES REPORTADAS EN EL 607/i.test(allText))formType='anexoa';

        else if(/ITBIS/i.test(allText))formType='it1';

    }


    const result={};const meta=extractMeta(allRows);

    if(formType==='ir17')parseIR17(allRows,result);

    else if(formType==='it1')parseIT1(allRows,result);

    else if(formType==='anexoa')parseAnexoA(allRows,result);

    else if(formType==='ir3')parseIR3(allRows,result);

    else if(['ir2','act','a1','b1','d','d1','d2','e','g','h1','h2','j'].includes(formType)){

        Object.assign(result,parseIR2Data(formType,allRows));

        Object.assign(result,meta);

    }


    return{data:result,formType,rawText:debugLog};

}

function buildSheet(formType,fieldsDef){

    const data=results.filter(r=>r&&r.formType===formType).sort((a,b)=>(a.data.periodo_fiscal||'').localeCompare(b.data.periodo_fiscal||''));

    if(!data.length)return null;

    const per=data.map(r=>r.data.periodo_fiscal||'?');

    const rows=[['Campo',...per]];

    for(const f of fieldsDef){

        if(f.isSection)rows.push([f.label,...per.map(()=>'')]);

        else{const row=[f.label];for(const r of data)row.push(r.data[f.id]??'');rows.push(row);}

    }

    const ws=XLSX.utils.aoa_to_sheet(rows);

    ws['!cols']=[{wch:60},...per.map(()=>({wch:22}))];

    return ws;

}
